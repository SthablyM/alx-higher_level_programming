import unittest
from your_module_directory.my_list_module import MyList
---------------------------------------------------------------------
---------------

class TestMyList(unittest.TestCase):
    def test_print_sorted(self):
	my_list = MyList()
        my_list.append(1)
        my_list.append(4)
        my_list.append(2)
        my_list.append(3)
        my_list.append(5)
	print(my_list)
	[1, 4, 2, 3, 5]
	my_list.print_sorted()
	[1, 2, 3, 4, 5]
	print(my_list)
	1, 4, 2, 3, 5]

        self.assertEqual(str(my_list), "[1, 4, 2, 3, 5]")

        my_list.print_sorted()
        self.assertEqual(str(my_list), "[1, 2, 3, 4, 5]")

if __name__ == '__main__':
    unittest.main()
